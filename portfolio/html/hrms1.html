Framework Snippet

using HRMSAutomation.Drivers;

namespace HRMSAutomation.Steps
{
    [Binding]
    public class SettingsPageStepDefinitions
    {
        private readonly ScenarioContext _scenarioContext;

        Pages page;

        public SettingsPageStepDefinitions(ScenarioContext scenarioContext)
        {
            page = scenarioContext.Get("Pages");
            _scenarioContext = scenarioContext;
        }


        [Then(@"Click The Employee List Button")]
        public void ThenClickTheEmployeeListButton()
        {
            page.SettingsPage.SettingsOrganizationPolicies.ClickEmployeeListButton();
            page.OrganizationPage.WaitForThePageLoadToComplete(false);
        }


        [Then(@"Validate The Added Policies To The Employees")]
        public void ThenValidateTheAddedPoliciesToTheEmployees()
        {
            page.SettingsPage.SettingsOrganizationPolicies.ValidateThePoliceForEmployees();
        }
    }

}
